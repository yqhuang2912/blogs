Python 示例 #

以下示例展示了一个包含注释、多行语句与 f-string 的 Python 代码块：

from pathlib import Path

# 加载数据并统计行数
def count_lines(paths: list[Path]) -> dict[str, int]:
    summary = {}
    for path in paths:
        text = path.read_text(encoding="utf-8")
        summary[path.name] = len(text.splitlines())
    return summary

if __name__ == "__main__":
    stats = count_lines([Path("notes.md"), Path("todo.md")])
    for name, lines in stats.items():
        print(f"{name}: {lines} 行")
